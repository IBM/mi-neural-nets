777  eT cosineSimilarity(eT *A, eT *B, size_t length_)
778  {
779  eT dot = 0.0, denom_a = 0.0, denom_b = 0.0 ;
780  for(size_t i = 0; i < length_; ++i) {
781  dot += A[i] * B[i] ;
782  denom_a += A[i] * A[i] ;
783  denom_b += B[i] * B[i] ;
784  }
785  return dot / (sqrt(denom_a) * sqrt(denom_b)) ;
786  }
